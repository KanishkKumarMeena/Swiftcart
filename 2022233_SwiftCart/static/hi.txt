<script>
  function updateQuantityDisplay(input) {
      var displayElement = document.getElementById('quantity-display-' + input.id.split('-')[1]);
      var sliderValueElement = document.getElementById('slider-value-' + input.id.split('-')[1]);
      displayElement.innerText = input.value;
      sliderValueElement.innerText = input.value; // Update value at which the slider is
  }
</script>
<script>
$(document).ready(function() {
    $('.btn-plus').on('click', function() {
        var itemId = $(this).data('field');
        var quantityInput = $('#quantity-' + itemId);
        var quantity = parseInt(quantityInput.val());
        var maxQuantity = parseInt(quantityInput.attr('max'));
        if (!isNaN(quantity) && quantity < maxQuantity) {
            quantityInput.val(quantity + 1);
        }
    });

    $('.btn-minus').on('click', function() {
        var itemId = $(this).data('field');
        var quantityInput = $('#quantity-' + itemId);
        var quantity = parseInt(quantityInput.val());
        if (!isNaN(quantity) && quantity > 0) {
            quantityInput.val(quantity - 1);
        }
    });
});
</script>
